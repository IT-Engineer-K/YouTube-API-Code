from apiclient.discovery import build
YOUTUBE_API_KEY = 'ここに取得したAPIキーを貼り付けます。'
youtube = build('youtube', 'v3', developerKey=YOUTUBE_API_KEY)
KeyWords = ['才能','コンピューター','神回']#こんな感じで検索キーワードを20個とか30個くらい記入します。※この時キーワードを''で囲み、その間を,で区切ります。
for k in range(len(KeyWords)):
    search_response = youtube.search().list(
        part='snippet',
        q=KeyWords[k],
        order='viewCount',
        type='video',
        maxResults = 50
        ).execute()
    for i in range(50):
        print(search_response["items"][i]["snippet"]["title"])
